// ══════════════════════════════════════════════════ // AGENDAMENTO GFB — Google Apps Script // Deploy como "App da Web": Executar como Eu, Qualquer pessoa acessa // ══════════════════════════════════════════════════ var CONFIG = { calendars: { marcelo: "dedelorodrigues@gmail.com", eduardo: "direcaouniclass@gmail.com", yves: "ssolucoesempresariais4@gmail.com" }, duracao_min: 90, horarios: ["09:00","10:30","14:30","15:30"], // 12:00 removido (almoço) janela_dias: 10 }; // ── Ponto de entrada GET: retorna disponibilidade ── function doGet(e) { try { var result = getDisponibilidade(); return jsonResponse(result); } catch(err) { return jsonResponse({ erro: err.message }); } } // ── Ponto de entrada POST: cria evento ── function doPost(e) { try { var payload = JSON.parse(e.postData.contents); var result = criarEvento(payload); return jsonResponse(result); } catch(err) { return jsonResponse({ erro: err.message }); } } // ── Retorna slots livres para cada consultor ── function getDisponibilidade() { var hoje = new Date(); hoje.setHours(0,0,0,0); var slots = {}; // { marcelo: [{iso, label}], eduardo: [...], yves: [...] } var pessoas = Object.keys(CONFIG.calendars); for (var i = 0; i < pessoas.length; i++) { var pessoa = pessoas[i]; var email = CONFIG.calendars[pessoa]; slots[pessoa] = getSlotsLivres(email, hoje, CONFIG.janela_dias); } return { ok: true, slots: slots, gerado: new Date().toISOString() }; } function getSlotsLivres(email, hoje, janelaDias) { var livres = []; var cal; try { cal = CalendarApp.getCalendarById(email); } catch(e) { return []; } if (!cal) return []; for (var d = 1; d <= janelaDias; d++) { var dia = new Date(hoje); dia.setDate(hoje.getDate() + d); var dow = dia.getDay(); // 0=Dom, 6=Sáb if (dow === 0 || dow === 6) continue; // pula fim de semana // Busca eventos do dia var inicio = new Date(dia); inicio.setHours(0,0,0,0); var fim = new Date(dia); fim.setHours(23,59,59,999); var eventos; try { eventos = cal.getEvents(inicio, fim); } catch(e) { continue; } // Verifica cada horário configurado for (var h = 0; h < CONFIG.horarios.length; h++) { var hora = CONFIG.horarios[h]; var partes = hora.split(':'); var hh = parseInt(partes[0]); var mm = parseInt(partes[1] || '0'); // Bloqueio quarta de manhã if (dow === 3 && hh < 12) continue; // Bloqueio almoço (12:00 em Seg/Qua/Sex) if (hh === 12 && (dow === 1 || dow === 3 || dow === 5)) continue; var slotInicio = new Date(dia); slotInicio.setHours(hh, mm, 0, 0); var slotFim = new Date(slotInicio.getTime() + CONFIG.duracao_min * 60 * 1000); // Verifica se ultrapassa 17h var limite = new Date(dia); limite.setHours(17, 0, 0, 0); if (slotFim > limite) continue; // Verifica conflito com eventos existentes var livre = true; for (var ev = 0; ev < eventos.length; ev++) { var evInicio = eventos[ev].getStartTime(); var evFim = eventos[ev].getEndTime(); // Sobreposição: não (evFim <= slotInicio OU evInicio >= slotFim) if (!(evFim <= slotInicio || evInicio >= slotFim)) { livre = false; break; } } if (livre) { var iso = Utilities.formatDate(slotInicio, 'America/Sao_Paulo', "yyyy-MM-dd'T'HH:mm:ss"); // label legível em pt-BR var nomesDia = ['domingo','segunda-feira','terça-feira','quarta-feira','quinta-feira','sexta-feira','sábado']; var meses = ['jan','fev','mar','abr','mai','jun','jul','ago','set','out','nov','dez']; var label = nomesDia[dow] + ', ' + ('0'+dia.getDate()).slice(-2) + ' de ' + meses[dia.getMonth()] + ' às ' + hora + ' (1h30)'; livres.push({ iso: iso, label: label, date: Utilities.formatDate(slotInicio,'America/Sao_Paulo','yyyy-MM-dd'), hora: hora }); } } } return livres; } // ── Cria evento na agenda do consultor ── function criarEvento(payload) { var email = CONFIG.calendars[payload.consultor]; if (!email) return { erro: 'Consultor não encontrado: ' + payload.consultor }; var cal; try { cal = CalendarApp.getCalendarById(email); } catch(e) { return { erro: 'Não foi possível acessar a agenda: ' + e.message }; } if (!cal) return { erro: 'Agenda não encontrada: ' + email }; // Parse do horário (vem como "2025-01-15T09:00:00" no fuso de SP) var inicio = new Date(payload.slot_inicio); var durMs = (CONFIG.duracao_min || 90) * 60 * 1000; var fim = new Date(inicio.getTime() + durMs); var titulo = 'Consultoria GFB — ' + (payload.nome || 'Lead'); var descr = [ '👤 Nome: ' + (payload.nome || '—'), '📱 WhatsApp: ' + (payload.whatsapp || '—'), '📧 E-mail: ' + (payload.email || '—'), '🏋️ Academia: ' + (payload.academia || '—'), '', '📊 Diagnóstico: ' + (payload.diagnostico || '—'), '', payload.relatorio || '' ].join('\n'); try { var evento = cal.createEvent(titulo, inicio, fim, { description: descr, guests: payload.email || '' }); return { ok: true, event_id: evento.getId(), event_link: evento.getOriginalCalendarId() ? 'https://calendar.google.com/calendar/event?eid=' + Utilities.base64Encode(evento.getId()) : null, inicio: inicio.toISOString(), fim: fim.toISOString() }; } catch(e) { return { erro: 'Erro ao criar evento: ' + e.message }; } } // ── Helper: resposta JSON com CORS ── function jsonResponse(obj) { var output = ContentService.createTextOutput(JSON.stringify(obj)); output.setMimeType(ContentService.MimeType.JSON); return output; }
